137 static inline CGFloat CGPointGetDistanceToRect(CGPoint p, CGRect r) {
138  r = CGRectStandardize(r);
139  if (CGRectContainsPoint(r, p)) return 0;
140  CGFloat distV, distH;
141  if (CGRectGetMinY(r) <= p.y && p.y <= CGRectGetMaxY(r)) {
142  distV = 0;
143  } else {
144  distV = p.y < CGRectGetMinY(r) ? CGRectGetMinY(r) - p.y : p.y - CGRectGetMaxY(r);
145  }
146  if (CGRectGetMinX(r) <= p.x && p.x <= CGRectGetMaxX(r)) {
147  distH = 0;
148  } else {
149  distH = p.x < CGRectGetMinX(r) ? CGRectGetMinX(r) - p.x : p.x - CGRectGetMaxX(r);
150  }
151  return MAX(distV, distH);
152 }

258  rect.origin.y + rect.size.height));
259  CGRect ret = CGRectMake(origin.x, origin.y, corner.x - origin.x, corner.y - origin.y);
